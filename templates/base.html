// Show loading animation on page transitions document.addEventListener('click', function(e) { if (e.target.tagName === 'A' && e.target.getAttribute('href') && !e.target.getAttribute('href').startsWith('#')) { const loader = document.getElementById('ai-loading'); loader.classList.remove('hidden'); setTimeout(() => { loader.classList.add('hidden'); }, 2000); } }); // Enhanced card animations with neon effect function animateCards() { document.querySelectorAll('.glass').forEach((card, index) => { card.style.opacity = '0'; card.style.transform = 'translateY(20px)'; card.offsetHeight; setTimeout(() => { card.style.transition = 'all 0.5s cubic-bezier(0.4, 0, 0.2, 1)'; card.style.opacity = '1'; card.style.transform = 'translateY(0)'; // Add hover effect card.addEventListener('mouseenter', () => { card.classList.add('neon-border'); }); card.addEventListener('mouseleave', () => { card.classList.remove('neon-border'); }); }, index * 100); }); } // Run animations window.addEventListener('load', animateCards); document.addEventListener('DOMContentLoaded', animateCards);
